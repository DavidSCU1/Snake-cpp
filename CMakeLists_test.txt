cmake_minimum_required(VERSION 3.16)
project(SVGTest)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(Qt6 REQUIRED COMPONENTS Core Widgets Svg)

qt_standard_project_setup()

qt_add_executable(SVGTest test_svg_simple.cpp)

qt_add_resources(SVGTest "resources" PREFIX "/" FILES
    images/spongebob_head.svg
    images/spongebob_body.svg
    images/krabby_patty.svg
    images/golden_spatula.svg
)

target_link_libraries(SVGTest Qt6::Core Qt6::Widgets Qt6::Svg)
